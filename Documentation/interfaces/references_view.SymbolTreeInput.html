Interface SymbolTreeInput<T>

A symbol tree input is the entry point for populating the references viewlet. Inputs must be anchored at a code location, they must have a title, and they must resolve to a model.

interface SymbolTreeInput<T> {
    contextValue: string;
    location: Location;
    title: string;
    resolve(): ProviderResult<SymbolTreeModel<T>>;
    with(location: Location): SymbolTreeInput<T>;
}
